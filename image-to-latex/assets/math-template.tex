% Aligned derivation
\begin{align*}
  f(x) &= (x + 1)^2 \\
       &= x^2 + 2x + 1 \\
       &\geq 0 \quad \forall x \in \mathbb{R}
\end{align*}

% Piecewise (cases)
\begin{equation*}
  \mathrm{ReLU}(x) = \begin{cases}
    x  & \text{if } x \geq 0, \\
    0  & \text{otherwise}.
  \end{cases}
\end{equation*}

% Matrix
\begin{equation*}
  A = \begin{pmatrix}
    a_{11} & \cdots & a_{1n} \\
    \vdots & \ddots & \vdots \\
    a_{m1} & \cdots & a_{mn}
  \end{pmatrix}
\end{equation*}

% Operator and summation
\begin{equation*}
  \mathrm{softmax}(\bm{z})_i
    = \frac{\exp(z_i)}{\sum_{j=1}^{n} \exp(z_j)}
\end{equation*}
